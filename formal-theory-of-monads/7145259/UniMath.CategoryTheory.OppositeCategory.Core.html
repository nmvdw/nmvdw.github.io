Definition opp_cat (C : category)
  : category.
Show proof.
  exists (opp_precat C).
  apply has_homsets_op.

Notation " C ^opp" := (opp_cat C) (at level 31).

Definition functor_op {C D : category} (F : functor C D)
  : functor (opp_cat C) (opp_cat D)
  := functor_opp F.
